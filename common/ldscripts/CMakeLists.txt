
set (IMAG_REGS_LD imag_regs.ld)
if (NOT EXISTS ${IMAG_REGS_LD})
    file(WRITE ${IMAG_REGS_LD} "# Imaginary registers\n")
    file(APPEND ${IMAG_REGS_LD} "# This file is automatically generated by CMakeLists.txt.  Do not edit.\n")
    set(_imag_reg 0)
    set(_last_imag_reg 0)
    while(${_imag_reg} LESS 256)
        set(_alternate "0")
        if(${_imag_reg} GREATER 0)
            set(_alternate "__rc${_last_imag_reg}+1")
        endif()
        file(APPEND ${IMAG_REGS_LD} "PROVIDE(__rc${_imag_reg} = DEFINED( __rc${_imag_reg} ) ? __rc${_imag_reg} : ${_alternate} )\n")
        set(_last_imag_reg ${_imag_reg})
        # Quaint cmake logic for incrementing a variable
        math(EXPR _imag_reg "${_imag_reg}+1")
    endwhile()
endif()
